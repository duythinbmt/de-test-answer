SELECT
  d.name AS "Department",
  e.name AS "Employee Name",
  e.salary AS "Salary"
FROM
  Employee e
  JOIN Department d ON e.departmentId = d.id
WHERE
  (
    SELECT
      count(DISTINCT salary)
    FROM
      Employee
    WHERE
      departmentID = d.id
      AND salary > e.salary
  ) < 3
ORDER BY
  d.id,
  e.salary DESC
